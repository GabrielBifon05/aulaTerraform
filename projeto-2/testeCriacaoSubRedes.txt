terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 4.16"
    }
  }

  required_version = ">= 1.2.0"
}

provider "aws" {
  region = "us-east-1"
}

//Criação de VPC
resource "aws_vpc" "vpc-labTerraform-sptech" {
  cidr_block = "10.0.0.0/16"
  enable_dns_support   = true
  enable_dns_hostnames = true
  tags = {
    Name = "vpc-labTerraform-sptech"
  }
}

//Criação de Subnet Pública
resource "aws_subnet" "sub-az1-pub-labTerraform-sptech" {
  vpc_id = aws_vpc.vpc-labTerraform-sptech.id
  cidr_block = "10.0.1.0/16"
  map_public_ip_on_launch = true
  availability_zone = "us-east-1a"
  tags = {
    Name = "sub-az1-pub-labTerraform-sptech"
  }
}

//Criação de Subnet Privada
resource "aws_subnet" "sub-az-pri-labTerraform-sptech" {
  vpc_id = aws_vpc.vpc-labTerraform-sptech.id
  cidr_block = "10.0.2.0/16"
  availability_zone = "us-east-1a"
  tags = {
    Name = "sub-az-pri-labTerraform-sptech"
  }
}

//Criação de Gateway
resource "aws_internet_gateway" "igw-labTerraform-sptech" {
  vpc_id = aws_vpc.vpc-labTerraform-sptech.id
  tags = {
    Name = "igw-labTerraform-sptech"
  }
}

//Criação de Nat Gateway (necessário para sub-rede privada acessar a internet)
resource "aws_eip" "nat-eip" {
    vpc = true  
}

resource "aws_nat_gateway" "nat" {
  allocation_id = aws_eip.nat-eip.id
  subnet_id = aws_subnet.sub-az1-pub-labTerraform-sptech.id
  tags = {
    Name = "nat"
  }
}

//Criação de Route Table Pública 
resource "aws_route_table" "rtb-labTerraform-sptech" {
  vpc_id = aws_vpc.vpc-labTerraform-sptech.id
  route = {
    cidr_block = "0.0.0.0/0"
    gateway_id = aws_internet_gateway.igw-igw-labTerraform-sptech.id
  }
  tags = {
    Name = "rtb-labTerraform-sptech"
  }
}

//Associação da Route Table com as Subnet Pública
resource "aws_route_table_association" "rtb-association-pub-labTerraform-sptech" {
  subnet_id = aws_subnet.sub-az1-pub-labTerraform-sptech.id
  route_table_id = aws_route_table_association.rtb-association-pri-labTerraform-sptech.id
}

//Associação da Route Table com as Subnet Privada
resource "aws_route_table_association" "rtb-association-pri-labTerraform-sptech" {
  subnet_id = aws_subnet.sub-az-pri-labTerraform-sptech.id
